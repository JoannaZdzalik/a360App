<!-- wyciągnij wszystkie pytania dla uczestnika o id = ? -->
SELECT question_text, question_type, default_answers 
FROM questions 
LEFT JOIN sessions_questions
ON questions.id = sessions_questions.id_question
LEFT JOIN sessions
ON sessions_questions.id_session = sessions.id
LEFT JOIN participants
ON sessions.id = participants.id_session
WHERE participants.id = ?;

<!-- wyciągnij wszystkie odpowiedzi dla uczestników z sesji w której end_date < now() -->
SELECT sessions.name, participants.email, questions.question_text, answers.answer_text
FROM sessions
LEFT JOIN participants
ON sessions.id = participants.id_session
LEFT JOIN answers
ON participants.id = answers.id_participant
LEFT JOIN questions
ON answers.id_question = questions.id
WHERE sessions.end_date < now() AND is_sent = 'false';

ON sessions_questions.id_session = sessions.id

<!-- wyciągnij odpowiedzi dla emailu ania z sesji 1 -->
SELECT question_text, answer_text, id_participant
FROM answers
LEFT JOIN questions
ON answers.id_question = questions.id
WHERE answers.id_participant = (SELECT id FROM participants WHERE id_session = 1 AND email LIKE 'anna%');
